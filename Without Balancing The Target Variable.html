# Set the bar labels
bar_labels = ['Decision Tree','Random_Forest','Logistic Regression','KNN','Naive Bayes','SVM','MLP']

# Create the x position of the bars and give bar graph parameters
N=7
x_pos = np.arange(N)

# Create the plot bars
accuracy_scores=(Decision_Tree_testing_accuracy,Random_Forest_testing_accuracy,Logistic_Regression_testing_accuracy,
                 KNN_testing_accuracy,Naive_Bayes_testing_accuracy,SVM_testing_accuracy,MLP_testing_accuracy)

#We can plot other performance factors like F1-measures along with Accuracy, therefore we can use subplots.
fig, ax =plt.subplots(figsize=(12,5))

# Plot a bar graph. 
plt.bar([p for p in x_pos],# using the data from the mean_values
        accuracy_scores, 
        color='orange'
        )

# Set the chart's title
ax.set_title('Classifier Models Comparison Graph')

# Set axes labels and title
ax.set_ylabel('Accuracy %')
ax.set_xticks([p for p in x_pos])
ax.set_xticklabels(bar_labels)
plt.ylim(0.1,1)
plt.legend(['Accuracy Scores'])
plt.show()
